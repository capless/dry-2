Simple to Use

Just drop in the components and customize with attributes

<!-- Include the library -->
<script src="https://unpkg.com/dry2-components/dist/bundle.js"></script>

<!-- Use components with simple attributes -->
<dry-drawer position="right" drawer-class="bg-white w-80">
    <span slot="header">My Drawer</span>
    <div slot="content">
        <p>Drawer content goes here!</p>
    </div>
</dry-drawer>

<avatar-component name="John Doe" size="lg"></avatar-component>

<toast-component type="success" auto-show>
    Operation completed successfully!
</toast-component>
